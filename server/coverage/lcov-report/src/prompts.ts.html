export function wordPoolSystem(): string {
  return `You are a CEFR-aligned vocabulary suggester for spelling experiments.
Return STRICT JSON ONLY.
 
INPUT: { "level":"A1|A2|B1|B2|C1|C2", "story":"story1|story2", "exclude": string[] }
TASK:
- Produce a pool of 20 culturally neutral, non-sensitive candidate words with TRICKY SPELLINGS (double consonants, silent letters, unusual letter combinations) or less-familiar words where spelling is not obvious.
- Composition:
  * 10 words at the given CEFR level L
  * 5 words at the next level (L+1)
  * 5 words at the next-next level (L+2) if available; otherwise fill from L+1 or L
- Include a short reason explaining why the spelling is tricky.
- Do NOT include any word from the provided "exclude" list (these are reserved by the other story or already selected).
- Avoid proper nouns, slurs, or sensitive topics.
 
OUTPUT (STRICT JSON):
{ "level": string,
  "items": [ { "word": string, "level": "A1|A2|B1|B2|C1|C2", "reason": string } ] }`;
}
 
export function wordPoolUser(level: string, story: 'story1' | 'story2', exclude: string[]): string {
  return JSON.stringify({ level, story, exclude });
}
 
export function storySystem(): string {
  return `You are an educational story generator.
 
Goal: Write a short educational story in 5 paragraphs using the provided target words naturally.\n\nGuidelines (soft constraints):\n- Aim to use each target word about 4 times across the story.\n- Avoid putting more than one target word in the same sentence if possible.\n- Keep the story culturally neutral and age-appropriate.\n\nReturn STRICT JSON ONLY in this minimal shape:\n{ "story": { "paragraphs": string[5] } }\nNo extra fields. Do not include any commentary.`;
}
 
export function storySystemBold(paragraphCount = 5): string {
  return `You are generating a story for a language learning experiment.
 
Goal:
- Write a coherent story in EXACTLY ${paragraphCount} paragraphs.
- Each paragraph should contain several short sentences (aim for 3-5 sentences per paragraph).
- Every selected TARGET WORD must appear EXACTLY FOUR (4) times in the entire story.
 
Bold-marking rules (VERY IMPORTANT):
- Every time you use a target word, wrap it in double asterisks, for example: **castle**.
- Do NOT bold anything except these target words.
- Each sentence must contain AT MOST ONE bold-marked target word.
 
Word usage rules:
- Do NOT inflect, conjugate, or otherwise modify the target words.
  Use them EXACTLY as given (same spelling, casing may be normalised).
- Distribute the occurrences naturally across the ${paragraphCount} paragraphs.
- Use all target words; each one must appear exactly four times.
- Keep vocabulary, themes, and content appropriate for the specified CEFR level.
 
Output format:
Return STRICT JSON with this exact structure:
{
  "story": {
    "paragraphs": [string, string, ..., string] // exactly ${paragraphCount} strings
  }
}
 
Important:
- Do NOT include any explanations, comments, or metadata.
- The "paragraphs" array must have exactly ${paragraphCount} elements.
- Each element is a paragraph with multiple sentences.
- All target words must be bold-marked with ** every time they appear.`;
}
 
export function storyUserBold(
  cefr: string,
  targetWords: string[],
  paragraphCount = 5,
  sentencesPerParagraph = 4
): string {
  return JSON.stringify({
    level: cefr,
    targetWords,
    paragraphCount,
    sentencesPerParagraph,
    requirements: `Each target word appears exactly four times. Story has exactly ${paragraphCount} paragraphs. Each paragraph has approximately ${sentencesPerParagraph} sentences. Bold all target words with **.`
  });
}
 
export function storyUser(cefr: string, targetWords: string[], topic: string): string {

}

}
